[docs]def infrecrate(zone):
    """

    :param zone: Metroton, Suburbium or will return a default
    :return:
    """
    if zone == 'Metroton':
        return [1e-4,1e-2,1e-5]
    elif zone == 'Suburbium':
        return [1e-3,1e-2,1e-4]
    else:
        return [1e-4,1e-2,1e-5]
